使用Jquery 实现视频的弹幕

效果图

js代码

<script type="text/javascript">
        $(function() {
            //enter 发送
            $(".txt").keypress(function(event) {
                if(event.keyCode == 13) {
                    //发送弹幕
                    send();
                    //清空输入框
                    reset_input();
                }
            })
            //按钮发送
            $(".btn").click(function() {
                send();
                //清空输入框
                reset_input();
            })

        });

        //点击弹幕按钮会发生的事情
        function move() {
            if($('.son').position().left == 0) {
                $("div .son").animate({
                    left: "40px",
                    backgroundColor: "#F66"
                }, 1000);
                $(".mess").hide();

            } else if($('.son').position().left == 40) {
                $("div .son").animate({
                    left: "0px",
                    backgroundColor: "green"
                }, 1000);
                $(".mess").show();
            }

        }

        function send() {
            var num_h = 40 + Math.random() * 350;
            $(".mess").append("<p style='top:" + num_h + "px'>" + $(".txt").val() + "</p>");
            var num_s = 5000 + Math.random() * 10000;
            $("p").animate({
                left: '-200px'
            }, num_s);
        }

        function reset_input() {
            $(".txt").val("");
        }
    </script>
